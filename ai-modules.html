Python Decorators

Decorators are like gift wrappers for functions...

def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"Function took {end - start} seconds")
        return result
    return wrapper

def factorial(n):
    if n <= 1:

    return n * factorial(n - 1)
